def returnFlowLimit(cfsValue: float, SAGBI_Class: str) -> float:
    """
    Returns the flow limit for give soil / flow rate
    :param cfsValue: cubic feet per second
    :param SAGBI_Class: soil classification
    :return: flow rate in acre-feet
    """

    # 1 CFS is 1.983 acrefeetdaily
    RATIO = 1.983

    # 5 CFS(all soils: excellent, good, moderately good)
    # 10 CFS for excellent and good.5 CFS for moderately good
    # 15 CFS for excellent, 10 CFS for good, 5 CFS for moderately good

    # values not in the dictionary have a 5 CFS default (ie maxFlowLimitBySAGBI.get('bad', 5) -> 5 )
    maxFlowLimitBySABGI = {"Moderately Good": 5, "Good": 10, "Excellent": 15}

    cfs = min(cfsValue, maxFlowLimitBySABGI.get(SAGBI_Class, 5))

    return cfs * RATIO

class BaseSite(object):
    def __init__(
        self,
        acreage,
        crop=None,
        r_type=None,
        pumpingCost=0,
        cost_pw=0,
        cost=0,
        cost_unused=0,
        capitalCost=0,
        perc_rate=0,
        capacity=0,
        bottleneck_id=None,
        **kwargs,
    ):
        """
        Base site class
        Args:
            acreage: acres
            crop: name of the crop
            r_type: recharge type
            pumpingCost: pumping cost per acre-foot
            cost_pw: cost per week
            cost: cost for using the site
            cost_unused: cost for not using the site
            capitalCost: total capital cost for recharge
            perc_rate: perculation rate
            capacity: total volume the site/location can hold
            bottleneck_id: restriction identifier
            **kwargs: all other arguments
        """

        # TODO: catch all extra kwargs provided to class instance
        # self.__dict__.update(kwargs) # catch all for all unknown kwargs

        # required inputs
        self.crop = crop if crop is not None else r_type
        self.acreage = acreage
        self.r_type = r_type

        # inputs that defaults
        self.pumpingCost = pumpingCost  #: pumping cost per acre-foot
        self.cost_pw = cost_pw
        self.cost = cost
        self.cost_unused = cost_unused
        self.capitalCost = capitalCost
        self.perc_rate = perc_rate
        self.capacity = capacity
        self.bottleneck_id = bottleneck_id  # TODO verify that this can be NONE

        # added instance attributes
        self.unusedVolume = np.full(365, 0.0)
        self.usedVolume = np.full(365, 0.0)
        self.precip = np.full(365, 0.0)
        self.sort_index: int = 0
        self.index: int = 0
        self.compatible_sources = []
        self.dailyRechargeVolumes = np.full(365, 0.0)
        self.currentCapacity: float = 0

    def getSiteCost(self):
        """
        Calculate the annual cost for the site depending on use. Site cost is the capital cost + cost per used day +
        annual cost (acreage times the start up cost or acreage time the cost for being unused) + the annual pumping
        cost.
        :return: annual cost of operating the site in dollars
        """
        numberOfTimesUsed = np.count_nonzero(self.usedVolume)
        annualPumpingCost = np.sum(self.usedVolume) * self.pumpingCost
        costPerUsedDay = numberOfTimesUsed * (self.cost_pw / 7) * self.acreage

        if numberOfTimesUsed > 0:
            annualCost = self.acreage * self.cost
        else:
            annualCost = self.acreage * self.cost_unused

        return self.capitalCost + costPerUsedDay + annualCost + annualPumpingCost

    def annualPrecip(self):
        """
        Sum up all of the daily precipitation measurements to return the annual precipitation
        :return: annual precipitation
        """
        return np.sum(self.precip)

    def annualAFRecharge(self):
        """
        Sums up the daily recharge volumes (acre-feet) for the site
        :return: annual recharge in acre-feet
        """
        return np.sum(self.dailyRechargeVolumes)

    def setDailyRecharge(self, newVolume: float, day: int):

        self.dailyRechargeVolumes[day] += newVolume

    def setUnused(self, volume: float, day: int):
        """
        sets the unused volume of a site for a specific day
        :param volume: new value to add to unusedVolume for a specific day
        :param day: index for specific day
        """
        self.unusedVolume[day] = volume

    def __repr__(self):
        return f"{self.__dict__}"
