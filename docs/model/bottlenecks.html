# bottlenecks
import numpy as np


class Bottleneck(object):
    def __init__(self, bottleneck_id=None, canal=None, flow_cfs=None):
        self.bottleneck_id = bottleneck_id
        self.canal = canal
        self.flow_cfs = flow_cfs
        self.AcreFeetPerDay = (
            flow_cfs * 1.983
        )  # cubic feet per second to acre-feet per day


class Bottlenecks(object):
    def __init__(self, bottleneck_list=[]):

        self.capacityHitLogger = [
            [] for _ in range(365)
        ]  # should this be a list of lists?
        self.bottleneck_list = bottleneck_list
        self.nameMap = dict()
        self.availableThroughputs = dict()

        # loop through the list of bottlenecks and add to dictionary (use bottleneck_id as dict keys)
        for b in self.bottleneck_list:
            self.nameMap[b.bottleneck_id] = b.canal
            self.availableThroughputs[b.bottleneck_id] = np.full(
                365, fill_value=b.AcreFeetPerDay
            )

    def updateUsage(self, bid: int, day: int, amount: float):
        """
        Update the bottleneck available throughput
        :param bid: bottleneck_id
        :param day: index date
        :param amount: usage volume in acre-feet to subtract
        :return:
        """
        self.availableThroughputs[bid][day] -= amount
        return

    def logger(self, bid: int, day: int):
        """
        Adds bottleneck id to the capacity hit logger
        :param bid: bottleneck id
        :param day: index day
        :return:
        """
        # If bNeck ID is already in the logger, skip it, we only care about the first time it gets in
        dayValueArray = self.capacityHitLogger[day]

        if bid not in dayValueArray:
            dayValueArray.append(bid)
        return

    def getAvailableThroughput(self, bid: int, day: int):
        """
        For a given day + bottleneck, return the throughput. If daily throughput is less than zero, add bottleneck_id
        to the bottlenecks logger for the day
        :param bid: bottleneck_id
        :param day: index date
        :return:
        """
        dailyThroughput = self.availableThroughputs[bid][day]
        if dailyThroughput <= 0:
            # capacity hit
            self.logger(bid, day)
        else:
            return dailyThroughput

    def getAll(self):
        return self.availableThroughputs

    def getLogs(self):
        return self.capacityHitLogger

    def names(self):
        return self.nameMap

    def __repr__(self):
        return f"{self.__dict__}"
