# basins
import numpy as np

from .baseSite import BaseSite


class Basin(BaseSite):
    def __init__(self, *args, **kwargs):
        super(Basin, self).__init__(*args, **kwargs)
        self.currentCapacity = 0

    def updateDailyValues(self, index: int):
        """
        Update daily recharge vol array's value based on index update the current capacity
        :param index: current day index
        :return:
        """
        rechargeVolume = min(self.perc_rate, self.currentCapacity)
        self.setDailyRecharge(rechargeVolume, index)
        self.currentCapacity = max(self.currentCapacity - self.perc_rate, 0)

    def getSiteCost(self):
        """
        Calculates the number of times that a site is used. If site is used (count of the number of times
        usedVolume > 0), returns the site cost. Otherwise, site is not used and should return the cost_unused
        :return: cost or cost_unused
        """
        # get count of number of times this site is used. The non-zero values in usedVolume numpy array
        # return site cost if number times used is above zero, else return cost_unused
        num_times_used = np.count_nonzero(self.usedVolume)
        return self.cost if num_times_used > 0 else self.cost_unused

    def getSiteVolume(self):
        """
        basin volume is the capacity minus the current capacity plus the perc rate
        :return:
        """
        return self.capacity - self.currentCapacity + self.perc_rate

    def setUsedVolume(self, newVolume: float, day: int):
        """
        set used volume for a specific date
        :param newVolume: new value to add to the used volume for the day and to the overall currentCapacity
        :param day: index for the current date
        :return:
        """
        self.usedVolume[day] += newVolume
        self.currentCapacity = self.currentCapacity + newVolume
        return

    def useSite(self, index: int, wafr, bottlenecks, dailyPrecip: float):
        """
        Use the basin
        :param index: day index
        :param wafr: water available for recharge
        :param bottlenecks: bottleneck obj
        :param dailyPrecip: precipitation
        :return:
        """

        # loop through the sources
        for src in self.compatible_sources:

            # get volumeAvailableFromWafr
            volumeAvailableFromWafr = wafr[src][index]

            # set precip
            self.setPrecip(dailyPrecip, index)

            # calc available site volume
            availableSiteVolume = max(
                self.getSiteVolume(), 0
            )  # TODO does this function have an index day param?

            # calc siteVolumeNeed
            siteVolumeNeeded = max(availableSiteVolume - dailyPrecip, 0)

            # calc availableThroughput
            availableThroughput = max(
                bottlenecks.getAvailableThroughput(self.bottleneck_id, index),
                0,
            )

            # calc wafrApplied
            # Use max WAFR that can be applied is the minimum value of limiting factors.
            wafrApplied = min(
                siteVolumeNeeded, volumeAvailableFromWafr, availableThroughput
            )

            # setUsedVolume
            self.setUsedVolume(wafrApplied, index)

            # subtract from wafr
            wafr[src][index] -= wafrApplied

            # update bottleneck usage
            bottlenecks.updateUsage(self.bottleneck_id, index, wafrApplied)

        # update daily values
        self.updateDailyValues(index)
        return
